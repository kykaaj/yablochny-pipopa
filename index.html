Здесь мы сделаем так, чтобы браузер запоминал, что ты прочитала, и нарисуем красивые "светлячки" на фоне. ```javascript // --- 1. Логика "Прочитано" --- // Эта функция запускается, когда сайт загрузился document.addEventListener('DOMContentLoaded', () => { // Проверяем каждую карточку const cards = document.querySelectorAll('.card'); cards.forEach(card => { const id = card.getAttribute('data-id'); // Если в памяти есть отметка, делаем фанфик серым if (localStorage.getItem(id) === 'true') { card.classList.add('read'); const btn = card.querySelector('.mark-btn'); if(btn) btn.textContent = 'Прочитано ✓'; } }); }); // Функция, которая срабатывает при нажатии кнопки function toggleRead(id) { const card = document.querySelector(`.card[data-id="${id}"]`); const btn = card.querySelector('.mark-btn'); // Если уже прочитано - убираем отметку if (localStorage.getItem(id) === 'true') { localStorage.removeItem(id); card.classList.remove('read'); btn.textContent = 'Отметить прочитанным'; } else { // Если не прочитано - ставим отметку и сохраняем в память localStorage.setItem(id, 'true'); card.classList.add('read'); btn.textContent = 'Прочитано ✓'; } } // --- 2. Красивые частицы на фоне (Canvas) --- const canvas = document.getElementById('canvas-bg'); const ctx = canvas.getContext('2d'); let particlesArray; // Растягиваем канвас на весь экран canvas.width = window.innerWidth; canvas.height = window.innerHeight; // Класс для частички class Particle { constructor() { this.x = Math.random() * canvas.width; this.y = Math.random() * canvas.height; this.size = Math.random() * 2; // Размер this.speedX = (Math.random() * 1) - 0.5; this.speedY = (Math.random() * 1) - 0.5; this.color = Math.random() > 0.5 ?
